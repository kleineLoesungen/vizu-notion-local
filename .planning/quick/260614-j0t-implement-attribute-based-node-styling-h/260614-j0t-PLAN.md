// Tableau 10 palette — 10 accessible, visually distinct colors.
// palette index helper: returns a hex color string by index, wrapping around if index >= 10.
// Called as {{palette @index}} inside {{#each (group ...)}} to auto-assign one color per group.
const PALETTE = [
  '#4e79a7', '#f28e2b', '#e15759', '#76b7b2', '#59a14f',
  '#edc948', '#b07aa1', '#ff9da7', '#9c755f', '#bab0ac',
]

Handlebars.registerHelper('palette', function(index: unknown) {
  const i = Math.abs(Number(index) || 0)
  return PALETTE[i % PALETTE.length]
})
```

No SafeString wrapper needed — hex strings are safe plain text and are used inside `classDef` lines, not as raw HTML.

  %% Example: attribute-based node styling with palette
  %% Each distinct value of "parent" gets a unique color automatically.
  %% classDef names must be valid Mermaid identifiers — use nodeId to derive a stable name.
  {{#each (group my-source "parent")}}
  classDef cls-{{nodeId "parent" parent}} fill:{{palette @index}},stroke:{{palette @index}},color:#fff,rx:5
  {{#group-item}}
  {{title}}:::cls-{{nodeId "parent" parent}}
  {{/group-item}}
  {{/each}}
```

Note: `{{nodeId "parent" parent}}` is written explicitly here (not `{{parent}}`) because inside a `classDef` line the value is used as a CSS class name suffix, not as a standalone node definition. The rewriter would turn `{{parent}}` into a full `nXXXXXX["label"]` string which is not valid as a class name.
