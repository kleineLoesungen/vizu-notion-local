    In `<script setup>`, add a `useDropdown` factory function (NOT exported — local to the script block) immediately after the existing imports. The factory creates an independent dropdown instance:

    ```ts
    function useDropdown() {
      const isOpen = ref(false)
      const containerRef = ref<HTMLElement | null>(null)
      const toggle = () => { isOpen.value = !isOpen.value }
      const close = () => { isOpen.value = false }
      const onDocClick = (e: MouseEvent) => {
        if (containerRef.value && !containerRef.value.contains(e.target as Node)) {
          close()
        }
      }
      onMounted(() => document.addEventListener('click', onDocClick, true))
      onBeforeUnmount(() => document.removeEventListener('click', onDocClick, true))
      return { isOpen, containerRef, toggle, close }
    }
    ```
